If Else If Condition

Condition if 
				const a = 10;
				if (a === 10) {
					alert(true);
				}
Boolean Conversion 
				if (0) { // 0 is false
				....
				}
			
				if (1) { // 1 is true
				....
				}
Else 
				const a = 10;
				if (a === 10) {
					alert(true);
				} else {
					alert(false)
				}
else if 
				const a = 10;
				if (a === 10) {
					alert("a = 10");
				} else if (a === 20) {
					alert("a = 20")
				} else {
					alert(false)
				}
Condition operator ? 
				let result = condition ? value1 : value2;
				example
				let accessAllowed = (age > 18) ? true : false;
			
				multiple
				let age = prompt('age?', 18);

				let message = (age < 3) ? 'Hi, baby!' :
				(age < 18) ? 'Hello!' :
				(age < 100) ? 'Greetings!' :
				'What an unusual age!';

				alert( message );
